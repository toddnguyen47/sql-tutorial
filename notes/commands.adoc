= SQL Commands

[cols="d, a", stripes=none]
|===
| Description | Command

| Show All Databases
|
[source, SQL]
----
SHOW DATABASES;
----

| Use a Database
|
[source, SQL]
----
Use database_name;
----

| Create a Database
|
[source, SQL]
----
CREATE TABLE tablename (
  column_name data_type,
  column_name data_type
);
----

| Show a Table. You'll need to select a DB first
|
[source, SQL]
----
SHOW TABLES;
----

| Show columns from a table
|
[source, SQL]
----
SHOW COLUMNS FROM table_name;

DESC table_name;
----

|===
